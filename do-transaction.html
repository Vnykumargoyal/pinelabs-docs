Overview init doTransaction

doTransaction

stable Since 0.1.0 transaction

Dispatch a Sale transaction to the connected terminal.

Description

doTransaction is the single entry point for payment transactions. It returns immediately; the outcome is delivered asynchronously to the supplied TransactionListener.

1
Generates a UUIDv4 eventId and invokes listener.onStarted(eventId).
2
Binds to the MasterApp service and sends the request as a JSON Message (bundle key MASTERAPPREQUEST, what = 1001).
3
Waits up to timeoutSecs (set at init) for a reply on a private Messenger.
4
Parses the JSON response envelope and routes to onSuccess or onFailure accordingly.
ℹ️
Only SALE is supported in v0.1.0. REFUND, VOID, PREAUTH, etc. will extend TransactionType in a later minor version.

Example

Kotlin 
import com.pinelabs.sdk.PinelabsSdk
import uniffi.pinelabs.*

val request = TransactionRequest(
    amount          = 9900uL,                          // ₹99.00
    billingRefNo    = "TX${System.currentTimeMillis()}",
    invoiceNo       = "INV-1001",
    transactionType = TransactionType.SALE,
)

sdk.doTransaction(request, object : TransactionListener {
    override fun onStarted(eventId: String) {
        Log.i("Pinelabs", "Transaction started: $eventId")
    }
    override fun onSuccess(result: TransactionResult) {
        Log.i("Pinelabs", "Approved — auth: ${result.authCode}, RRN: ${result.rrn}")
    }
    override fun onFailure(error: SdkException) {
        when (error) {
            is SdkException.TransportUnavailable -> Log.e("Pinelabs", "Terminal unavailable: ${error.detail}")
            is SdkException.Timeout              -> Log.w("Pinelabs", "Timed out — verify transaction status")
            is SdkException.TransactionFailed    -> Log.w("Pinelabs", "Declined (${error.terminalResponseCode}): ${error.detail}")
            is SdkException.InvalidInput         -> Log.e("Pinelabs", "Bad request: ${error.detail}")
            is SdkException.Internal             -> Log.e("Pinelabs", "SDK error: ${error.detail}")
            is SdkException.NotInitialized       -> Log.e("Pinelabs", "SDK not initialised")
        }
    }
})
Java 
import com.pinelabs.sdk.PinelabsSdk;
import uniffi.pinelabs.*;

long   amount      = 9900L;                              // ₹99.00
String billingRef  = "TX" + System.currentTimeMillis();
String invoiceNo   = "INV-1001";

TransactionRequest request = new TransactionRequest(
    amount,
    billingRef,
    invoiceNo,
    TransactionType.SALE
);

sdk.doTransaction(request, new TransactionListener() {
    @Override
    public void onStarted(String eventId) {
        Log.i("Pinelabs", "Transaction started: " + eventId);
    }
    @Override
    public void onSuccess(TransactionResult result) {
        Log.i("Pinelabs", "Approved — auth: " + result.getAuthCode() + ", RRN: " + result.getRrn());
    }
    @Override
    public void onFailure(SdkException error) {
        if (error instanceof SdkException.TransportUnavailable e) {
            Log.e("Pinelabs", "Terminal unavailable: " + e.getDetail());
        } else if (error instanceof SdkException.Timeout) {
            Log.w("Pinelabs", "Timed out — verify transaction status");
        } else if (error instanceof SdkException.TransactionFailed e) {
            Log.w("Pinelabs", "Declined (" + e.getTerminalResponseCode() + "): " + e.getDetail());
        } else if (error instanceof SdkException.InvalidInput e) {
            Log.e("Pinelabs", "Bad request: " + e.getDetail());
        } else if (error instanceof SdkException.Internal e) {
            Log.e("Pinelabs", "SDK error: " + e.getDetail());
        } else {
            Log.e("Pinelabs", "Not initialised — call PinelabsSdk.init() first");
        }
    }
});
