Detecting the device at runtime

Use the constants from com.pinelabs.sdk.devices.DeviceModels to branch on the current terminal:

Kotlin 
import android.os.Build
import com.pinelabs.sdk.devices.DeviceModels.*

when (Build.MODEL) {
    INGENICO_DX8000              -> { /* Ingenico DX8000 specific path */ }
    KOZEN_P3, KOZEN_L200         -> { /* Kozen */ }
    WISEASY_P052, WISEASY_P5,
    WISEASY_T2                   -> { /* Wiseasy */ }
    PAX_A60, PAX_A80, PAX_A920,
    PAX_A920_PRO, PAX_A910,
    PAX_A910S, PAX_A950, PAX_A77,
    PAX_A50, PAX_A50D, PAX_A99,
    PAX_IM30                     -> { /* PAX */ }
    N950, N950S                  -> { /* Newland */ }
    else                         -> { /* Unsupported model */ }
}
